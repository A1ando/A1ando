Python小程序（2）身份证号码校验并提取信息

2020-11-26 python小程序 ， 原创 ， python

身份证号码主要包括以下知识点：

  • 地区码：按照GB/T2260执行，通常1开头为华北地区、2开头为东北地区、3开头为华东地区、4开头为华中地区和华南地区、5开头为西南地区、6开头为西北地区、7和8开头为特别地区。

  • 出生日期码：按GB/T7408执行

  • 顺序码：表示同一地址码所标识的区域范围内，对同年同月出生的人编订的顺序号，奇数分为男性、偶数分给女性

  • 校验码：按照ISO 7064:1983.MOD11-2计算出来的校验码得到第18位数字

    校验方法：详见validate(ID)函数

import pandas as pd
from datetime import datetime
import bisect


flag = False

def validate(ID):
    global flag
    validate_seq = list('10x98765432')
    factor_seq = [7, 9, 10, 5, 8, 4, 2, 1, 6, 3, 7, 9, 10, 5, 8, 4, 2]

    if len(ID) == 18:
        try:
            index = sum(map(lambda x, y: int(x) * y, list(ID[:17]), factor_seq)) % 11
            if ID[-1].lower() == validate_seq[index]:
                flag = True
        except ValueError:
            flag = False
    return flag


def get_area(ID):
    global flag
    birth = ID[0:6]
    path = r'全国行政区域身份证代码表(utf-8-sig).csv'
    df = pd.read_csv(path, encoding='utf-8-sig')
    if int(birth) >= 110101:  # 没有这个判断，数值低于110101的地区码会显示为‘北京市市辖区东城区’
        try:
            index = bisect.bisect_left(df['areacode'], int(birth))
            return df['nativeplace'][index]
        except KeyError:
            pass
    else:
        return flag


def get_birth(ID):
    global flag
    birth = ID[6:14]
    date_from = datetime(year=1900, month=1, day=1)  # 1900-01-01 00:00:00
    date_to = datetime.today()
    birth_date = datetime(year=int(birth[:4]), month=int(birth[4:6]), day=int(birth[6:]))

    if date_from < birth_date <= date_to:
        age = date_to.year - birth_date.year
        birthday = birth_date.strftime('%Y-%m-%d')
        return {'age': age, 'birthday': birthday}
    else:
        return flag


def get_gender(ID):
    if int(ID[16]) % 2 == 0:  # 第17位是偶数为女性，奇数为男性
        gender = '女'
    else:
        gender = '男'
    return gender


def run(lst):
    for ID in lst:
        ID = str(ID)
        # 全部校验flag,任何一处有问题说明身份证号无效
        if validate(ID) and get_area(ID) and get_birth(ID):  
            yield {'身份证号': ID,
                   '户籍所在地': get_area(ID),
                   '出生日期': get_birth(ID)['birthday'],
                   '年龄': get_birth(ID)['age'],
                   '性别': get_gender(ID)}
        else:
            print(f'{ID} 身份证号无效！')


if __name__ == '__main__':
    # 从网上找一些身份证号码验证
    ID_lst = [130532197901235712, 513221197102183838,
              500101198401133397, 431128198405258384, 150900198904133964,
              '44092319850718266X', '36100119860426330X']
    for i in run(ID_lst):
        print(i)
